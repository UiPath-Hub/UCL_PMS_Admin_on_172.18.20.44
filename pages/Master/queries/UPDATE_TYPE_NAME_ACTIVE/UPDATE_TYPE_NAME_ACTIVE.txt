DECLARE @ACTIVEVERSION nvarchar(20);
SET @ACTIVEVERSION = (SELECT TOP 1 [VERSION] FROM dbo.PMS_MASTER_LIST WHERE TYPE_NAME={{appsmith.store["TYPE_NAME_MODAL"]}} AND [ACTIVE] = 1);

UPDATE dbo.PMS_MASTER_LIST SET ACTIVE=0 WHERE TYPE_NAME={{appsmith.store["TYPE_NAME_MODAL"]}} AND VERSION=@ACTIVEVERSION;

UPDATE dbo.PMS_MASTER_LIST SET ACTIVE=1 WHERE TYPE_NAME={{appsmith.store["TYPE_NAME_MODAL"]}} AND VERSION={{appsmith.store["CurrentVersion"].VERSION}};
